```js
// https://leetcode-cn.com/problems/hua-dong-chuang-kou-de-zui-da-zhi-lcof/

/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number[]}
 */
var maxSlidingWindow = function (nums, k) {
  var res = []
  if (k === 0) return []
  for (var i = 0; i <= nums.length - k; i++) {
    var max = nums[i]
    for (var j = 1; j < k; j++) {
      max = Math.max(max, nums[i + j])
    }
    res.push(max)
  }

  return res
};
```
